import { db, collection, getDocs, query, where } from "./firebase-public.js"; const grid = document.getElementById("productGrid"); async function loadProducts() { const q = query( collection(db, "products"), where("status", "==", "active") ); const snap = await getDocs(q); grid.innerHTML = ""; snap.forEach(d => { const p = d.data(); const card = document.createElement("div"); card.className = "card"; card.innerHTML = `
${p.name}
${p.summary || ""}
Lihat Produk `; grid.appendChild(card); }); } loadProducts();
